76 template <typename S>
77 ObjectPtr<S> FclGroup<S>::create_object(math::GeometryPtr shape)
78 {
79  if (!shape)
80  {
81  dtwarn
82  << "Not allowed to create a collision object for a shape of nullptr.\n";
83  return nullptr;
84  }
85 
86  auto fcl_collision_geometry
87  = createFclCollisionGeometry<S>(shape, FclEngine<S>::MESH);
88  if (!fcl_collision_geometry)
89  {
90  dtwarn << "Failed to create FCL collision geometry.\n";
91  return nullptr;
92  }
93 
94  return std::shared_ptr<FclObject<S>>(
95  new FclObject<S>(this, std::move(shape), fcl_collision_geometry));
96 }
