58 Eigen::Matrix<S, 3, 3> Sphere<S>::computeInertiaFromMass(S radius, S mass) {
59  Eigen::Matrix<S, 3, 3> inertia = Eigen::Matrix<S, 3, 3>::Zero();
60 
61  inertia(0, 0) = S(2) / S(5) * mass * radius * radius;
62  inertia(1, 1) = inertia(0, 0);
63  inertia(2, 2) = inertia(0, 0);

65  return inertia;
