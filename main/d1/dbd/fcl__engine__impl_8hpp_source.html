81 bool FclEngine<S>::collide(ObjectPtr<S> object1, ObjectPtr<S> object2) {
82  auto derived1 = std::dynamic_pointer_cast<FclObject<S>>(object1);
83  if (!derived1) {
84  dterr << "Invalid object\n";
85  return false;
86  }
87 
88  auto derived2 = std::dynamic_pointer_cast<FclObject<S>>(object2);
89  if (!derived2) {
90  dterr << "Invalid object\n";
91  return false;
92  }
93 
94  auto fcl_collision_object1 = derived1->get_fcl_collision_object();
95  if (!fcl_collision_object1) {
96  dterr << "Invalid object\n";
97  return false;
98  }
99 
100  auto fcl_collision_object2 = derived2->get_fcl_collision_object();
101  if (!fcl_collision_object2) {
102  dterr << "Invalid object\n";
103  return false;
104  }
105 
106  FclCollisionRequest<S> request;
107  FclCollisionResult<S> result;
108 
109  return ::fcl::collide(
110  fcl_collision_object1, fcl_collision_object2, request, result);
111 }

121 template <typename S>
122 std::shared_ptr<FclCollisionGeometry<S>>
123 FclEngine<S>::create_fcl_collision_geometry_impl(
124  const math::ConstGeometryPtr& shape, FclEngine<S>::PrimitiveShape type) {
125  FclCollisionGeometry<S>* geom = nullptr;
126  const auto& shapeType = shape->getType();
127 
128  if (auto sphere = shape->as<math::Sphered>()) {
129  const auto radius = sphere->getRadius();
130 
131  if (FclEngine<S>::PRIMITIVE == type) {
132  geom = new FclSphere<S>(radius);
133  } else {
134  auto fcl_mesh = new ::fcl::BVHModel<FclOBBRSS<S>>();
135  auto fcl_sphere = FclSphere<S>(radius);
136  ::fcl::generateBVHModel(
137  *fcl_mesh, fcl_sphere, FclTransform3<S>(), 16, 16);
138  geom = fcl_mesh;
139  }
140  } else {
141  dterr << "[FclEngine<S>::createFCLCollisionGeometry] "
142  << "Attempting to create an unsupported shape type [" << shapeType
143  << "]. Creating a sphere with 0.1 radius "
144  << "instead.\n";
145 
146  geom = new FclSphere<S>(0.1);
147  }
148 
149  assert(geom);
150 
151  return std::shared_ptr<FclCollisionGeometry<S>>(geom);
152 }
