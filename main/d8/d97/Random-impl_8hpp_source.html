318 template <typename S>
319 ::Eigen::Quaternion<S> Random::uniformUnitQuaternion() {
320  // "Uniform Random Rotations" of Shoemake:
321  // http://planning.cs.uiuc.edu/node198.html
322 
323  static_assert(
324  std::is_floating_point_v<S>, "Non-floating point type is not supported");
325 
326  constexpr S pi = constants<S>::pi();
327 
328  const S u1 = uniform<S>(0, 1);
329  const S u2 = uniform<S>(0, 1);
330  const S u3 = uniform<S>(0, 1);
331 
332  const S a = std::sqrt(1. - u1);
333  const S b = S(2) * pi * u2;
334  const S c = std::sqrt(u1);
335  const S d = S(2) * pi * u3;
336 
337  return ::Eigen::Quaternion<S>(
338  a * std::sin(b), a * std::cos(b), c * std::sin(d), c * std::cos(d));
339 }
