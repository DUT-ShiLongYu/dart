105 template <typename... Others>
106 void CollisionGroup::subscribeTo(
107  const dynamics::ConstBodyNodePtr& bodyNode, const Others&... others) {
108  const auto inserted = mBodyNodeSources.insert(BodyNodeSources::value_type(
109  bodyNode.get(), BodyNodeSource(bodyNode.get(), bodyNode->getVersion())));
110 
111  if (inserted.second) {
112  const BodyNodeSources::iterator& entry = inserted.first;
113 
114  const auto collisionShapeNodes
115  = bodyNode->getShapeNodesWith<dynamics::CollisionAspect>();
116 
117  for (const auto& shapeNode : collisionShapeNodes) {
118  entry->second.mObjects.insert(
119  {shapeNode, addShapeFrameImpl(shapeNode, bodyNode.get())});
120  }
121  }
122 
123  subscribeTo(others...);
124 }

127 template <typename... Others>
128 void CollisionGroup::subscribeTo(
129  const dynamics::ConstSkeletonPtr& skeleton, const Others&... others) {
130  const auto inserted = mSkeletonSources.insert(SkeletonSources::value_type(
131  skeleton.get(), SkeletonSource(skeleton, skeleton->getVersion())));
132 
133  if (inserted.second) {
134  SkeletonSource& entry = inserted.first->second;
135 
136  const std::size_t numBodies = skeleton->getNumBodyNodes();
137  for (std::size_t i = 0u; i < numBodies; ++i) {
138  const dynamics::BodyNode* bn = skeleton->getBodyNode(i);
139 
140  const auto& collisionShapeNodes
141  = bn->getShapeNodesWith<dynamics::CollisionAspect>();
142 
143  auto& childInfo
144  = entry.mChildren
145  .insert(std::make_pair(
146  bn, SkeletonSource::ChildInfo(bn->getVersion())))
147  .first->second;
148 
149  for (const auto& shapeNode : collisionShapeNodes) {
150  entry.mObjects.insert(
151  {shapeNode, addShapeFrameImpl(shapeNode, skeleton.get())});
152  childInfo.mFrames.insert(shapeNode);
153  }
154  }
155  }
156 
157  subscribeTo(others...);
158 }
